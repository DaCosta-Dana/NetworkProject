function Sender {
    send_base = 0
    nextseqnum = 0

    while True {
        if nextseqnum < send_base + window_size{
            send packet nextseqnum
            nextseqnum ++
        }

        if receiveAck for send_base {
            send_base++
            if send_base == nextseqnum {
                stop timer
            } else {
                start timer (measures the timeout for the packet at send_base)
            }
            if timeout {
                start timer
                send packet send_base
                send packet send_base
                ...
                send packet nextseqnum-1
            }
        }
    }
}
